В вашем распоряжении имеется 2 одинаковых стеклянных шарика и n-этажное здание. Требуется определить минимальное количество бросков, достаточное для точного определения того этажа, начиная с которого шарики будут разбиваться.

int getNumberOfAttempts(int n) {
    var best = new int[n + 1];
    for (var i = 1; i <= n; ++i) {
        best[i] = i;
        for (var j = i - 1; j > 0; --j) {
            best[i] = Math.Min(best[i], 1 + Math.Max(i - j, best[j - 1]));
        }
    }
    return best[n];
}
